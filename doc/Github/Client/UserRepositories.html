def valid_user?(repo, options = {})
  response = get(repo.public_users_path, options)
  if response[:id].present?
    true
  else
    raise Github::UserNotFound, [NO_USER,
                                 '[',
                                 response[:message].to_s,
                                 ']'].join
  end
end
